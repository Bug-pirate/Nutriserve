:: Step 1: Create project folder and enter it
mkdir express-app
cd express-app

:: Step 2: Initialize npm project
npm init -y

:: Step 3: Install express
npm install express

:: Step 4: Create folder structure
mkdir src
cd src
mkdir routes controllers models services middlewares config utils
cd ..

:: Step 5: Create main entry file and other boilerplate files
echo // Entry point > src\index.js
echo // User routes > src\routes\userRoutes.js
echo // User controller > src\controllers\userController.js
echo // User model > src\models\userModel.js
echo // User service > src\services\userService.js
echo // Auth middleware > src\middlewares\authMiddleware.js
echo // DB config > src\config\db.js
echo // Helper functions > src\utils\helpers.js

:: Step 6: Create .env and .gitignore
echo PORT=3000 > .env
echo node_modules/ > .gitignore

:: Step 7: Create README
echo # Express App > README.md
